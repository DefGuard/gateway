85fn add_address_to_set(set: *mut nftnl_sys::nftnl_set, ip: &Address) -> Result<(), FirewallError> {
86    match ip {
87        Address::Network(network) => {
88            let upper_bound = max_address(network);
89            let net = network.network();
90            match (net, upper_bound) {
91                (IpAddr::V4(network), IpAddr::V4(upper_bound)) => {
92                    add_to_set(set, &network, Some(&upper_bound))?;
93                }
94                (IpAddr::V6(network), IpAddr::V6(upper_bound)) => {
95                    add_to_set(set, &network, Some(&upper_bound))?;
96                }
97                _ => {
98                    return Err(FirewallError::InvalidConfiguration(format!(
99                        "Expected both addresses to be of the same type, got {net:?} and \
100                        {upper_bound:?}",
101                    )))
102                }
103            }
104        }
105        Address::Range(addr_range) => match addr_range {
106            IpAddrRange::V4(ipv4_range) => {
107                add_to_set(set, ipv4_range.start(), Some(ipv4_range.end()))?;
108            }
109            IpAddrRange::V6(ipv6_range) => {
110                add_to_set(set, ipv6_range.start(), Some(ipv6_range.end()))?;
111            }
112        },
113    }
114
115    Ok(())
116}

118fn add_port_to_set(set: *mut nftnl_sys::nftnl_set, port: &Port) -> Result<(), FirewallError> {
119    match port {
120        Port::Any => {
121            // nothing to do
122        }
123        Port::Single(port) => {
124            let inet_service = InetService(*port);
125            add_to_set(set, &inet_service, Some(&inet_service))?;
126        }
127        Port::Range(start, end) => {
128            let start = InetService(*start);
129            let end = InetService(*end);
130
131            add_to_set(set, &start, Some(&end))?;
132        }
133    }
134
135    Ok(())
136}

142    add_to_set(set, proto, None)?;
143    Ok(())

146impl FirewallRule for FilterRule<'_> {
147    fn to_chain_rule<'a>(
148        &self,
149        chain: &'a Chain,
150        batch: &mut Batch,
151    ) -> Result<Rule<'a>, FirewallError> {
152        let mut rule = Rule::new(chain);
153        debug!("Converting {self:?} to nftables expression");
154        // Debug purposes only
155        let mut matches = Vec::new();
156
157        if !self.dest_ports.is_empty() && self.protocols.len() > 1 {
158            return Err(FirewallError::InvalidConfiguration(format!(
159                "Cannot specify multiple protocols with destination ports, specified \
160                protocols: {:?}, destination ports: {:?}, Defguard Rule ID: {}",
161                self.protocols, self.dest_ports, self.defguard_rule_id
162            )));
163        }
164
165        // TODO: Reduce code duplication here
166        if !self.src_ips.is_empty() {
167            if self.v4 {
168                let set = new_anon_set::<Ipv4Addr>(chain.get_table(), ProtoFamily::Inet, true)?;
169                batch.add(&set, nftnl::MsgType::Add);
170
171                for ip in self.src_ips {
172                    add_address_to_set(set.as_ptr(), ip)?;
173                }
174
175                // ip saddr {x.x.x.x, x.x.x.x}
176                set.elems_iter().for_each(|elem| {
177                    batch.add(&elem, nftnl::MsgType::Add);
178                });
179
180                rule.add_expr(&nft_expr!(meta nfproto));
181                rule.add_expr(&nft_expr!(cmp == libc::NFPROTO_IPV4 as u8));
182                rule.add_expr(&nft_expr!(payload ipv4 saddr));
183
184                rule.add_expr(&nft_expr!(lookup & set));
185            } else {
186                let set = new_anon_set::<Ipv6Addr>(chain.get_table(), ProtoFamily::Inet, true)?;
187                batch.add(&set, nftnl::MsgType::Add);
188
189                for ip in self.src_ips {
190                    add_address_to_set(set.as_ptr(), ip)?;
191                }
192
193                // ip6 saddr {x.x.x.x, x.x.x.x}
194                set.elems_iter().for_each(|elem| {
195                    batch.add(&elem, nftnl::MsgType::Add);
196                });
197
198                rule.add_expr(&nft_expr!(meta nfproto));
199                rule.add_expr(&nft_expr!(cmp == libc::NFPROTO_IPV6 as u8));
200                rule.add_expr(&nft_expr!(payload ipv6 saddr));
201
202                rule.add_expr(&nft_expr!(lookup & set));
203            }
204            debug!(
205                "Added source IP addresses match to nftables expression: {:?}",
206                self.src_ips
207            );
208            matches.push(format!("ANY SOURCE IPs: {:?}", self.src_ips));
209        }
210
211        // TODO: Reduce code duplication here
212        if !self.dest_ips.is_empty() {
213            if self.v4 {
214                let set = new_anon_set::<Ipv4Addr>(chain.get_table(), ProtoFamily::Inet, true)?;
215                batch.add(&set, nftnl::MsgType::Add);
216
217                for ip in self.dest_ips {
218                    add_address_to_set(set.as_ptr(), ip)?;
219                }
220
221                set.elems_iter().for_each(|elem| {
222                    batch.add(&elem, nftnl::MsgType::Add);
223                });
224
225                // ip daddr {x.x.x.x, x.x.x.x}
226                rule.add_expr(&nft_expr!(meta nfproto));
227                rule.add_expr(&nft_expr!(cmp == libc::NFPROTO_IPV4 as u8));
228                rule.add_expr(&nft_expr!(payload ipv4 daddr));
229
230                rule.add_expr(&nft_expr!(lookup & set));
231            } else {
232                let set = new_anon_set::<Ipv6Addr>(chain.get_table(), ProtoFamily::Inet, true)?;
233                batch.add(&set, nftnl::MsgType::Add);
234
235                for ip in self.dest_ips {
236                    add_address_to_set(set.as_ptr(), ip)?;
237                }
238
239                // ip6 daddr {x.x.x.x, x.x.x.x}
240                set.elems_iter().for_each(|elem| {
241                    batch.add(&elem, nftnl::MsgType::Add);
242                });
243
244                rule.add_expr(&nft_expr!(meta nfproto));
245                rule.add_expr(&nft_expr!(cmp == libc::NFPROTO_IPV6 as u8));
246                rule.add_expr(&nft_expr!(payload ipv6 daddr));
247
248                rule.add_expr(&nft_expr!(lookup & set));
249            }
250            debug!(
251                "Added destination IP addresses match to nftables expression: {:?}",
252                self.dest_ips
253            );
254            matches.push(format!("ANY DEST IPs: {:?}", self.dest_ips));
255        }
256
257        if !self.protocols.is_empty() {
258            // > 0 Protocols
259            // 0 Ports
260            if self.protocols.len() > 1 {
261                let set = new_anon_set::<Protocol>(chain.get_table(), ProtoFamily::Inet, false)?;
262                batch.add(&set, nftnl::MsgType::Add);
263
264                for proto in &self.protocols {
265                    add_protocol_to_set(set.as_ptr(), proto)?;
266                }
267
268                // <protocol> dport {x, x-x}
269                set.elems_iter().for_each(|elem| {
270                    batch.add(&elem, nftnl::MsgType::Add);
271                });
272
273                rule.add_expr(&nft_expr!(meta nfproto));
274
275                if self.v4 {
276                    rule.add_expr(&nft_expr!(cmp == libc::NFPROTO_IPV4 as u8));
277                    rule.add_expr(&nft_expr!(payload ipv4 protocol));
278                } else {
279                    rule.add_expr(&nft_expr!(cmp == libc::NFPROTO_IPV6 as u8));
280                    rule.add_expr(&nft_expr!(payload ipv6 nextheader));
281                }
282
283                rule.add_expr(&nft_expr!(lookup & set));
284
285                debug!("Added protocol match to rule: {:?}", self.protocols);
286                matches.push(format!("ANY PROTOCOLS: {:?}", self.protocols));
287            }
288            // 1 Protocol
289            // > 0 Ports
290            else if !self.dest_ports.is_empty() {
291                if let Some(protocol) = self.protocols.first() {
292                    if protocol.supports_ports() {
293                        let set = new_anon_set::<InetService>(
294                            chain.get_table(),
295                            ProtoFamily::Inet,
296                            true,
297                        )?;
298                        batch.add(&set, nftnl::MsgType::Add);
299
300                        for port in self.dest_ports {
301                            add_port_to_set(set.as_ptr(), port)?;
302                        }
303
304                        // <protocol> dport {x, x-x}
305                        set.elems_iter().for_each(|elem| {
306                            batch.add(&elem, nftnl::MsgType::Add);
307                        });
308
309                        rule.add_expr(&nft_expr!(meta l4proto));
310                        rule.add_expr(&nft_expr!(cmp == *protocol as u8));
311                        rule.add_expr(protocol.as_port_payload_expr()?);
312                        rule.add_expr(&nft_expr!(lookup & set));
313                    }
314                }
315
316                debug!(
317                    "Added single protocol ({:?}) match and destination ports match to nftables \
318                    expression: {:?}",
319                    self.protocols, self.dest_ports
320                );
321                matches.push(format!(
322                    "PROTOCOL: {:?} AND ANY DEST PORTS: {:?}",
323                    self.protocols, self.dest_ports
324                ));
325            }
326            // 1 Protocol
327            // 0 Ports
328            else if let Some(protocol) = self.protocols.first() {
329                // ip protocol <protocol>
330                rule.add_expr(&nft_expr!(meta nfproto));
331
332                if self.v4 {
333                    rule.add_expr(&nft_expr!(cmp == libc::NFPROTO_IPV4 as u8));
334                    rule.add_expr(&nft_expr!(payload ipv4 protocol));
335                } else {
336                    rule.add_expr(&nft_expr!(cmp == libc::NFPROTO_IPV6 as u8));
337                    rule.add_expr(&nft_expr!(payload ipv6 nextheader));
338                }
339
340                rule.add_expr(&nft_expr!(cmp == *protocol as u8));
341                debug!("Added protocol match to rule: {protocol:?}");
342                matches.push(format!("SINGLE PROTOCOL: {protocol:?}"));
343            }
344        }
345
346        if let Some(iifname) = &self.iifname {
347            // iifname <interface>
348            rule.add_expr(&nft_expr!(meta iifname));
349            let exact = InterfaceName::Exact(CString::new(iifname.as_str()).unwrap());
350            if self.negated_iifname {
351                rule.add_expr(&nft_expr!(cmp != exact));
352            } else {
353                rule.add_expr(&nft_expr!(cmp == exact));
354            }
355            debug!("Added input interface match to rule: {iifname:?}");
356            matches.push(format!("INPUT INTERFACE: {iifname:?}"));
357        }
358
359        if let Some(oifname) = &self.oifname {
360            // oifname <interface>
361            rule.add_expr(&nft_expr!(meta oifname));
362            let exact = InterfaceName::Exact(CString::new(oifname.as_str()).unwrap());
363            if self.negated_oifname {
364                rule.add_expr(&nft_expr!(cmp != exact));
365            } else {
366                rule.add_expr(&nft_expr!(cmp == exact));
367            }
368            debug!("Added output interface match to rule: {oifname:?}");
369            matches.push(format!("OUTPUT INTERFACE: {oifname:?}"));
370        }
371
372        if !self.states.is_empty() {
373            // ct state <state1>,<state2>
374            let combined_states = self
375                .states
376                .iter()
377                .fold(0u32, |acc, state| acc | state.to_expr_state().bits());
378            rule.add_expr(&nft_expr!(ct state));
379            rule.add_expr(&nft_expr!(bitwise mask combined_states, xor 0u32));
380            rule.add_expr(&nft_expr!(cmp != 0u32));
381            debug!(
382                "Added connection tracking states match to nftables expression: {:?}",
383                self.states
384            );
385            matches.push(format!("ANY CT STATES: {:?}", self.states));
386        }
387
388        if self.counter {
389            // counter
390            rule.add_expr(&nft_expr!(counter));
391            debug!("Added counter expression to rule");
392        }
393
394        // accept/drop
395        match self.action {
396            Policy::Allow => {
397                rule.add_expr(&nft_expr!(verdict accept));
398            }
399            Policy::Deny => {
400                rule.add_expr(&nft_expr!(verdict drop));
401            }
402        }
403
404        // comment <comment>
405        if let Some(comment_string) = &self.comment {
406            debug!("Adding comment to nftables expression: {comment_string:?}");
407            // Since we are interoping with C, truncate the string to 255 *bytes* (not UTF-8 characters)
408            // 256 is the maximum length of a comment string in nftables, leave 1 byte for the null terminator
409            let maybe_truncated_str = if comment_string.len() > 255 {
410                warn!("Comment string {comment_string} is too long, truncating to 255 bytes");
411                &comment_string[..=255]
412            } else {
413                comment_string.as_str()
414            };
415            let comment = &CString::new(maybe_truncated_str).map_err(|e| {
416                FirewallError::NetlinkError(format!(
417                    "Failed to create CString from string {comment_string}. Error: {e:?}"
418                ))
419            })?;
420            rule.set_comment(comment);
421            debug!("Added comment to nftables expression: {comment_string:?}");
422        } else {
423            debug!("No comment provided for nftables expression");
424        }
425
426        let matches = matches.join(" AND ");
427        debug!("Created nftables rule with matches: {matches:?}");
428
429        Ok(rule)
430    }
431}

434struct NatRule {
435    src_ip: Option<IpAddr>,
436    dest_ip: Option<IpAddr>,
437    oifname: Option<String>,
438    iifname: Option<String>,
439    negated_oifname: bool,
440    negated_iifname: bool,
441    counter: bool,
442}
443
444impl FirewallRule for NatRule {
445    fn to_chain_rule<'a>(
446        &self,
447        chain: &'a Chain,
448        _batch: &mut Batch,
449    ) -> Result<Rule<'a>, FirewallError> {
450        let mut rule = Rule::new(chain);
451
452        if let Some(src_ip) = self.src_ip {
453            if src_ip.is_ipv4() {
454                rule.add_expr(&nft_expr!(payload ipv4 saddr));
455            } else {
456                rule.add_expr(&nft_expr!(payload ipv6 saddr));
457            }
458            rule.add_expr(&nft_expr!(cmp == src_ip));
459        }
460
461        if let Some(dest_ip) = self.dest_ip {
462            if dest_ip.is_ipv4() {
463                rule.add_expr(&nft_expr!(payload ipv4 daddr));
464            } else {
465                rule.add_expr(&nft_expr!(payload ipv6 daddr));
466            }
467            rule.add_expr(&nft_expr!(cmp == dest_ip));
468        }
469
470        if let Some(iifname) = &self.iifname {
471            rule.add_expr(&nft_expr!(meta iifname));
472            let exact = InterfaceName::Exact(CString::new(iifname.as_str()).unwrap());
473            if self.negated_iifname {
474                rule.add_expr(&nft_expr!(cmp != exact));
475            } else {
476                rule.add_expr(&nft_expr!(cmp == exact));
477            }
478        }
479
480        if let Some(oifname) = &self.oifname {
481            rule.add_expr(&nft_expr!(meta oifname));
482            let exact = InterfaceName::Exact(CString::new(oifname.as_str()).unwrap());
483            if self.negated_oifname {
484                rule.add_expr(&nft_expr!(cmp != exact));
485            } else {
486                rule.add_expr(&nft_expr!(cmp == exact));
487            }
488        }
489
490        if self.counter {
491            rule.add_expr(&nft_expr!(counter));
492        }
493
494        rule.add_expr(&nft_expr!(masquerade));
495
496        Ok(rule)
497    }
498}

543/// Sets up the default chains for the firewall
544pub(super) fn init_firewall(
545    initial_policy: Policy,
546    defguard_fwd_chain_priority: Option<i32>,
547    batch: &mut Batch,
548    ifname: &str,
549) -> Result<(), FirewallError> {
550    let table = Tables::Defguard(ProtoFamily::Inet).to_table(ifname);
551
552    batch.add(&table, nftnl::MsgType::Add);
553    batch.add(&table, nftnl::MsgType::Del);
554    batch.add(&table, nftnl::MsgType::Add);
555
556    let mut fw_chain = Chains::Forward.to_chain(&table);
557    fw_chain.set_hook(
558        nftnl::Hook::Forward,
559        defguard_fwd_chain_priority.unwrap_or(FORWARD_PRIORITY),
560    );
561    fw_chain.set_policy(initial_policy.into());
562    fw_chain.set_type(nftnl::ChainType::Filter);
563    batch.add(&fw_chain, nftnl::MsgType::Add);
564
565    Ok(())
566}

589/// Applies masquerade on the specified interface for the outgoing packets
590pub(super) fn set_masq(
591    ifname: &str,
592    enabled: bool,
593    batch: &mut Batch,
594) -> Result<(), FirewallError> {
595    let table = Tables::Defguard(ProtoFamily::Inet).to_table(ifname);
596    batch.add(&table, nftnl::MsgType::Add);
597
598    drop_chain(&Chains::Postrouting, batch, ifname)?;
599
600    let mut nat_chain = Chains::Postrouting.to_chain(&table);
601    nat_chain.set_hook(nftnl::Hook::PostRouting, POSTROUTING_PRIORITY);
602    nat_chain.set_policy(nftnl::Policy::Accept);
603    nat_chain.set_type(nftnl::ChainType::Nat);
604    batch.add(&nat_chain, nftnl::MsgType::Add);
605
606    let nat_rule = NatRule {
607        oifname: Some(LOOPBACK_IFACE.to_string()),
608        negated_oifname: true,
609        counter: true,
610        ..Default::default()
611    }
612    .to_chain_rule(&nat_chain, batch)?;
613
614    if enabled {
615        batch.add(&nat_rule, nftnl::MsgType::Add);
616    } else {
617        batch.add(&nat_rule, nftnl::MsgType::Del);
618    }
619
620    Ok(())
621}

740pub(super) fn apply_filter_rules(
741    rules: Vec<FilterRule>,
742    batch: &mut Batch,
743    ifname: &str,
744) -> Result<(), FirewallError> {
745    let table = Tables::Defguard(ProtoFamily::Inet).to_table(ifname);
746    batch.add(&table, nftnl::MsgType::Add);
747
748    let forward_chain = Chains::Forward.to_chain(&table);
749    batch.add(&forward_chain, nftnl::MsgType::Add);
750
751    for rule in rules.iter() {
752        let chain_rule = rule.to_chain_rule(&forward_chain, batch)?;
753        batch.add(&chain_rule, nftnl::MsgType::Add);
754    }
755
756    Ok(())
757}

759pub(crate) fn send_batch(batch: &FinalizedBatch) -> Result<(), FirewallError> {
760    let socket = mnl::Socket::new(mnl::Bus::Netfilter)
761        .map_err(|e| FirewallError::NetlinkError(format!("Failed to create socket: {e:?}")))?;
762    socket.send_all(batch).map_err(|e| {
763        FirewallError::NetlinkError(format!("Failed to send batch through socket: {e:?}"))
764    })?;
765
766    let portid = socket.portid();
767    let mut buffer = vec![0; nftnl::nft_nlmsg_maxsize() as usize];
768
769    // TODO: Why is it supposed to be 2?
770    let seq = 2;
771    while let Some(message) = socket_recv(&socket, &mut buffer[..])? {
772        match mnl::cb_run(message, seq, portid) {
773            Ok(mnl::CbResult::Stop) => {
774                debug!("Received stop signal from netlink callback");
775                break;
776            }
777            Ok(mnl::CbResult::Ok) => {
778                debug!("Received OK signal from netlink callback");
779            }
780            Err(err) => {
781                return Err(FirewallError::NetlinkError(format!(
782                    "There was an error while sending netlink messages: {err:?}"
783                )))
784            }
785        };
786    }
787
788    Ok(())
789}

791fn socket_recv<'a>(
792    socket: &mnl::Socket,
793    buf: &'a mut [u8],
794) -> Result<Option<&'a [u8]>, FirewallError> {
795    let ret = socket.recv(buf).map_err(|err| {
796        FirewallError::NetlinkError(format!(
797            "Failed while reading a message from socket: {err:?}"
798        ))
799    })?;
800    if ret > 0 {
801        Ok(Some(&buf[..ret]))
802    } else {
803        Ok(None)
804    }
805}

807fn new_anon_set<T>(
808    table: &Table,
809    family: ProtoFamily,
810    interval_set: bool,
811) -> Result<Set<T>, FirewallError>
812where
813    T: SetKey,
814{
815    let set = Set::<T>::new(ANON_SET_NAME, get_set_id(), table, family);
816
817    if interval_set {
818        unsafe {
819            nftnl_sys::nftnl_set_set_u32(
820                set.as_ptr(),
821                nftnl_sys::NFTNL_SET_FLAGS as u16,
822                (libc::NFT_SET_ANONYMOUS | libc::NFT_SET_CONSTANT | libc::NFT_SET_INTERVAL) as u32,
823            );
824        }
825    }
826
827    Ok(set)
828}

830/// Adds key to a set. If the range_end option is specified, it will assume the lower and upper
831/// bounds of a range need to be added.
832fn add_to_set<K>(
833    set: *mut nftnl_sys::nftnl_set,
834    key: &K,
835    range_end: Option<&K>,
836) -> Result<(), FirewallError>
837where
838    K: SetKey,
839{
840    let key_data = key.data();
841    let key_data_len = key_data.len() as u32;
842    unsafe {
843        let elem = nftnl_sys::nftnl_set_elem_alloc();
844        if elem.is_null() {
845            return Err(FirewallError::OutOfMemory(
846                "Failed to allocate memory for set element".to_string(),
847            ));
848        }
849        nftnl_sys::nftnl_set_elem_set(
850            elem,
851            nftnl_sys::NFTNL_SET_ELEM_KEY as u16,
852            key_data.as_ptr().cast(),
853            key_data_len,
854        );
855        nftnl_sys::nftnl_set_elem_add(set, elem);
856
857        if let Some(end) = range_end {
858            let mut end_data = end.data();
859
860            // This is a workaround to make the upper bound inclusive.
861            // Perhaps there is a better way to do this.
862            increment_bytes(&mut end_data);
863            let end_data_len = (end_data.len()) as u32;
864
865            let elem = nftnl_sys::nftnl_set_elem_alloc();
866            if elem.is_null() {
867                return Err(FirewallError::OutOfMemory(
868                    "Failed to allocate memory for set element".to_string(),
869                ));
870            }
871            nftnl_sys::nftnl_set_elem_set(
872                elem,
873                nftnl_sys::NFTNL_SET_ELEM_KEY as u16,
874                end_data.as_ptr().cast(),
875                end_data_len,
876            );
877            nftnl_sys::nftnl_set_elem_set_u32(
878                elem,
879                nftnl_sys::NFTNL_SET_ELEM_FLAGS as u16,
880                libc::NFT_SET_ELEM_INTERVAL_END as u32,
881            );
882            nftnl_sys::nftnl_set_elem_add(set, elem);
883        }
884    }
885
886    Ok(())
887}

889fn increment_bytes(bytes: &mut [u8]) {
890    for i in (0..bytes.len()).rev() {
891        if bytes[i] < 255 {
892            bytes[i] += 1;
893            return;
894        } else {
895            bytes[i] = 0;
896        }
897    }
898
899    // the bytes have overflown, but that's okay for our purposes
900}
